    "## Cell 3: Define Preprocessing Pipeline\n",
    "\n",
    "**Scientific Reference**: Koelstra et al. (2011, IEEE TAC)\n",
    "\n",
    "Standard DEAP preprocessing steps:\n",
    "1. **Notch filter @ 50Hz**: Remove power line interference (AC mains)\n",
    "2. **Bandpass filter 0.5-50Hz**: Isolate EEG frequency range\n",
    "3. **Common Average Reference (CAR)**: Reduce volume conduction artifacts\n",
    "4. **Baseline removal**: Zero-mean each trial\n",
    "\n",
    "**Tools**: MNE-Python implements standard filters validated in literature"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": null,

    "def preprocess_eeg_trial(eeg_signal, sfreq=512, l_freq=0.5, h_freq=50, notch_freq=50):\n",
    "    \"\"\"\n",
    "    Apply standard DEAP preprocessing to single trial.\n",
    "    \n",
    "    Reference: Koelstra et al. (2011)\n",
    "    - Input: (32 EEG channels, 8064 samples @ 512Hz)\n",
    "    - Steps:\n",
    "      1. Notch @ 50Hz (power line removal)\n",
    "      2. Bandpass 0.5-50Hz (EEG frequency range)\n",
    "      3. CAR (Common Average Reference)\n",
    "      4. Baseline removal\n",
    "    \n",
    "    Args:\n",
    "        eeg_signal: (n_channels, n_samples) EEG data\n",
    "        sfreq: Sampling frequency (default 512 Hz)\n",
    "        l_freq: Low cutoff frequency (default 0.5 Hz)\n",
    "        h_freq: High cutoff frequency (default 50 Hz)\n",
    "        notch_freq: Notch frequency (default 50 Hz, AC mains)\n",
    "    \n",
    "    Returns:\n",
    "        preprocessed: (n_channels, n_samples) filtered data\n",
    "    \"\"\"\n",
    "    n_channels, n_samples = eeg_signal.shape\n",
    "    \n",
    "    # Create MNE Info object (required for filtering)\n",
    "    # Using standard 10-20 montage (first 19 channels are EEG, rest ignored)\n",
    "    ch_names = [f'EEG{i:02d}' for i in range(32)]\n",
    "    ch_types = ['eeg'] * 32\n",
    "    info = mne.create_info(ch_names=ch_names, sfreq=sfreq, ch_types=ch_types)\n",
    "    \n",
    "    # Create Raw object (required for MNE filtering)\n",
    "    raw = mne.io.RawArray(eeg_signal, info)\n",
    "    \n",
    "    # Step 1: Notch filter @ 50Hz (Koelstra et al. 2011)\n",
    "    raw.notch_filter(freqs=notch_freq, verbose=False)\n",
    "    \n",
    "    # Step 2: Bandpass filter 0.5-50Hz (Koelstra et al. 2011)\n",
    "    raw.filter(l_freq=l_freq, h_freq=h_freq, verbose=False)\n",
    "    \n",
    "    # Get filtered data\n",
    "    filtered_data = raw.get_data()\n",
    "    \n",
    "    # Step 3: Common Average Reference (CAR) - Koelstra et al. 2011\n",
    "    # Subtract mean across channels to reduce volume conduction\n",
    "    car_data = filtered_data - filtered_data.mean(axis=0, keepdims=True)\n",
    "    \n",
    "    # Step 4: Baseline removal (zero-mean per trial)\n",
    "    baseline_data = car_data - car_data.mean(axis=1, keepdims=True)\n",
    "    \n",
    "    return baseline_data\n",
    "\n",
    "print(\"✓ Preprocessing pipeline defined\")\n",
    "print(\"  Steps (Koelstra et al. 2011):\")\n",
    "print(\"    1. Notch @ 50Hz (power line)\")\n",
    "print(\"    2. Bandpass 0.5-50Hz (EEG range)\")\n",
    "print(\"    3. CAR (volume conduction reduction)\")\n",
    "print(\"    4. Baseline removal (zero-mean)\")"
   ]

    "## Cell 4: Preprocess All Subjects (LOSO Strategy)\n",
    "\n",
    "**Validation Strategy**: Leave-One-Subject-Out (LOSO)\n",
    "- Standard for cross-subject generalization (Koelstra et al. 2011, CBraMod Wang et al. ICLR 2025)\n",
    "- Process: For each subject as test → train on remaining 31 subjects\n",
    "- Here: Preprocess all 32 subjects independently"

    "# Create output directory\n",
    "output_dir = Path('./preprocessed_data')\n",
    "output_dir.mkdir(exist_ok=True)\n",
    "\n",
    "print(\"\\n[Preprocessing] All 32 DEAP subjects...\")\nprint(f\"Output: {output_dir.absolute()}\")\nprint(f\"\\nLOSO Strategy: Processing each subject independently\")\nprint(f\"(Train on 31 others, test on 1 - validated at next stage)\\n\")\n",
    "\n",
    "# Process all subjects\n",
    "preprocessed_subjects = {}\n",
    "\n",
    "for subject_id in range(1, 33):  # 32 subjects\n",
    "    try:\n",
    "        # Load raw DEAP\n",
    "        eeg_raw, labels = load_deap_subject(subject_id)\n",
    "        \n",
    "        # Extract only EEG channels (0-31)\n",
    "        eeg_eeg_only = eeg_raw[:, :32, :]  # Exclude channels 32-39 (other modalities)\n",
    "        \n",
    "        # Preprocess all 40 trials\n",
    "        preprocessed_trials = []\n",
    "        for trial_idx in range(eeg_eeg_only.shape[0]):\n",
    "            trial_data = eeg_eeg_only[trial_idx]  # (32 channels, 8064 samples)\n",
    "            preprocessed = preprocess_eeg_trial(trial_data)\n",
    "            preprocessed_trials.append(preprocessed)\n",
    "        \n",
    "        preprocessed_eeg = np.array(preprocessed_trials)  # (40, 32, 8064)\n",
    "        \n",
    "        # Extract valence & arousal (columns 0, 1 of labels)\n",
    "        valence = labels[:, 0]\n",
    "        arousal = labels[:, 1]\n",
    "        \n",
    "        # Convert to binary (threshold 5.0) - Koelstra et al. 2011 standard\n",
    "        valence_binary = (valence >= 5).astype(int)  # 0=low, 1=high\n",
    "        arousal_binary = (arousal >= 5).astype(int)  # 0=low, 1=high\n",
    "        \n",
    "        # Save preprocessed data\n",
    "        output_file = output_dir / f'subject_{subject_id:02d}_preprocessed.npz'\n",
    "        np.savez_compressed(\n",
    "            output_file,\n",
    "            eeg=preprocessed_eeg,  # (40, 32, 8064)\n",
    "            valence=valence_binary,  # (40,)\n",
    "            arousal=arousal_binary,  # (40,)\n",
    "            sfreq=512,\n",
    "            n_channels=32,\n",
    "            trial_duration_sec=eeg_eeg_only.shape[2] / 512\n",
    "        )\n",
    "        \n",
    "        preprocessed_subjects[subject_id] = {\n",
    "            'shape': preprocessed_eeg.shape,\n",
    "            'valence_counts': np.bincount(valence_binary),\n",
    "            'arousal_counts': np.bincount(arousal_binary)\n",
    "        }\n",
    "        \n",
    "        print(f\"✓ S{subject_id:02d}: {preprocessed_eeg.shape} | \"\n",
    "              f\"Valence: {preprocessed_subjects[subject_id]['valence_counts']} | \"\n",
    "              f\"Arousal: {preprocessed_subjects[subject_id]['arousal_counts']}\")\n",
    "    \n",
    "    except Exception as e:\n",
    "        print(f\"✗ S{subject_id:02d}: {e}\")\n",
    "\n",
    "print(f\"\\n✓ Preprocessing complete: {len(preprocessed_subjects)}/32 subjects\")"
